1from jina import Document, Flow
2from jina.types.document.generators import from_files
3import numpy as np
4
5flow = (Flow()
6        .add(uses="jinahub+docker://ImageTorchEncoder")
7        .add(uses="jinahub+docker://SimpleIndexer"))
8
9with flow:
10    images = []
11    for doc in from_files('fashion_images/*.png'):
12        doc.convert_image_uri_to_blob()
13        doc.blob = doc.blob.astype(np.uint8)
14        images.append(doc)
15    flow.index(images)
16
17    query = Document(uri='t-shirt.png')
18    query.convert_image_uri_to_blob()
19    query.blob = doc.blob.astype(np.uint8)
20    flow.search(query)
